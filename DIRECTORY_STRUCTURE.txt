Setup Commands:
===============
# Automated setup (recommended)
./fix-and-setup.sh                    # Linux/Mac
fix-and-setup.bat                     # Windows

# Manual setup
mv frontend/srs frontend/src
mv frontend/index-3.html frontend/index.html
cd frontend && npm install
cd ../backend && npm install
cp backend/env.example backend/.env

# Start application
cd frontend && npm run dev            # Frontend on port 5173
cd backend && npm start               # Backend on port 3001
